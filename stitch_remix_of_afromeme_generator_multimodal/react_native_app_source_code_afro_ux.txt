/**
 * SCREEN 2: ContextScreen
 */
export const ContextScreen = ({ navigation }) => {
  const [text, setText] = useState('');
  const [loading, setLoading] = useState(false);
  const [mood, setMood] = useState('Clash');

  const handleGenerate = () => {
    setLoading(true);
    setTimeout(() => {
      setLoading(false);
      navigation.navigate('Result', { type: 'text', content: text });
    }, 2000);
  };

  return (
    <SafeAreaView style={styles.container}>
      <Header title="Analyseur de Ndoki" onBack={() => navigation.goBack()} />
      <ScrollView contentContainerStyle={styles.scrollPadding}>
        <View style={styles.inputContainer}>
          <TextInput
            multiline
            numberOfLines={10}
            style={styles.textArea}
            placeholder="Colle ou écris l'échange ici... (Ex: Ce que ma copine m'a dit ce matin...)"
            placeholderTextColor={COLORS.textSecondary}
            value={text}
            onChangeText={setText}
          />
          <Text style={styles.charCount}>{text.length}/2000</Text>
        </View>

        <Text style={styles.label}>CHOISIR LE NDOKI (MOOD)</Text>
        <View style={styles.moodGrid}>
          {['Clash / Mbindi', 'Ndolo / Amour', 'Nyanga / Fierté', 'Sarcasme'].map((m) => (
            <TouchableOpacity 
              key={m} 
              onPress={() => setMood(m)}
              style={[styles.moodItem, mood === m && styles.moodItemActive]}
            >
              <Text style={[styles.moodText, mood === m && styles.moodTextActive]}>{m}</Text>
            </TouchableOpacity>
          ))}
        </View>

        {loading ? (
          <ActivityIndicator size="large" color={COLORS.primary} style={{ marginTop: 40 }} />
        ) : (
          <AfroButton title="Générer le Mème Textuel" onPress={handleGenerate} />
        )}
      </ScrollView>
    </SafeAreaView>
  );
};

/**
 * SCREEN 3: VoiceScreen
 */
export const VoiceScreen = ({ navigation }) => {
  const [status, setStatus] = useState('ready'); // ready, recording, captured
  
  const requestMicPermission = async () => {
    if (Platform.OS === 'android') {
      try {
        const granted = await PermissionsAndroid.request(
          PermissionsAndroid.PERMISSIONS.RECORD_AUDIO,
          {
            title: "Permission Microphone",
            message: "L'application a besoin d'accéder au micro pour transformer ta voix en meme.",
            buttonNeutral: "Plus tard",
            buttonNegative: "Annuler",
            buttonPositive: "OK"
          }
        );
        if (granted === PermissionsAndroid.RESULTS.GRANTED) {
          setStatus('recording');
        }
      } catch (err) {
        console.warn(err);
      }
    }
  };

  return (
    <SafeAreaView style={styles.container}>
      <Header title="La Voix du Kwatt" onBack={() => navigation.goBack()} />
      <View style={styles.voiceCenter}>
        <TouchableOpacity 
          style={[styles.micButton, status === 'recording' && styles.micButtonActive]}
          onPress={status === 'ready' ? requestMicPermission : () => setStatus('captured')}
        >
          <Text style={{fontSize: 40, color: COLORS.white}}>🎤</Text>
          {status === 'recording' && <View style={styles.waveOverlay} />}
        </TouchableOpacity>
        <Text style={styles.timer}>00:00</Text>
        <Text style={styles.voiceInstruction}>
          {status === 'ready' ? 'Appuie pour commencer le ndem audio.' : 
           status === 'recording' ? 'On t\'écoute... Appuie pour arrêter.' : 
           'Audio capturé ! Transcription IA en cours...'}
        </Text>
        
        {status === 'captured' && (
           <View style={styles.miniPlayer}>
              <TouchableOpacity><Text>▶️ Play</Text></TouchableOpacity>
              <TouchableOpacity onPress={() => setStatus('ready')}><Text>🗑️ Effacer</Text></TouchableOpacity>
           </View>
        )}
      </View>
      <View style={styles.footer}>
        <AfroButton 
          title="Envoyer l'Audio à l'IA" 
          color={COLORS.secondary} 
          onPress={() => navigation.navigate('Result', { type: 'voice' })}
        />
      </View>
    </SafeAreaView>
  );
};

/**
 * SCREEN 4: RemixerScreen
 */
export const RemixerScreen = ({ navigation }) => {
  const [image, setImage] = useState(null);
  
  const requestGalleryPermission = async () => {
    if (Platform.OS === 'android') {
      const permission = Platform.Version >= 33 
        ? PermissionsAndroid.PERMISSIONS.READ_MEDIA_IMAGES 
        : PermissionsAndroid.PERMISSIONS.READ_EXTERNAL_STORAGE;
        
      const granted = await PermissionsAndroid.request(permission);
      if (granted === PermissionsAndroid.RESULTS.GRANTED) {
        // Simuler sélection
        setImage(true);
      }
    }
  };

  return (
    <SafeAreaView style={styles.container}>
      <Header title="Remix de Statut" onBack={() => navigation.goBack()} />
      <ScrollView contentContainerStyle={styles.scrollPadding}>
        <View style={styles.previewContainer}>
          {!image ? (
            <View style={styles.emptyPreview}>
               <Text style={{fontSize: 48, marginBottom: 10}}>🖼️</Text>
               <Text style={styles.emptyText}>Aucun média sélectionné. Capture l'instant ou choisis dans ta galerie.</Text>
            </View>
          ) : (
            <View style={[styles.emptyPreview, { backgroundColor: '#ddd' }]}>
               <Text>Image Sélectionnée</Text>
            </View>
          )}
        </View>

        <View style={styles.row}>
          <TouchableOpacity style={styles.secBtn} onPress={requestGalleryPermission}>
            <Text style={styles.secBtnText}>Galerie</Text>
          </TouchableOpacity>
          <TouchableOpacity style={styles.secBtn}>
            <Text style={styles.secBtnText}>Photo</Text>
          </TouchableOpacity>
        </View>

        <View style={styles.inputGroup}>
           <Text style={styles.label}>EXPRESSION CAMEROUNAISE (OPTIONNEL)</Text>
           <TextInput style={styles.shortInput} placeholder="Ex: Ndem, Akié, C'est le feu..." />
        </View>

        <AfroButton title="Laisser Gemini Analyser" onPress={() => navigation.navigate('Result', { type: 'remix' })} />
      </ScrollView>
    </SafeAreaView>
  );
};
